#include <Arduino.h> 
#include <Wire.h> 
#include <Adafruit_GFX.h> 
#include <Adafruit_SSD1306.h> 
#include <ArduinoJson.h>  

#define SCREEN_WIDTH 128  //screen width of the ssd1306
#define SCREEN_HEIGHT 64  //screen height of the ssd1306
#define LED_PIN 2 //LED pin 

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1); 

// RTOS Handles 
QueueHandle_t commandQueue; 
TaskHandle_t t_Input;  //to take input 
TaskHandle_t t_Display; //DisplayTask
TaskHandle_t t_Blink;  //blink LED
// Shared Global Variable 
int currentDelay = 1000; // Default 1 second blink 
// Data Structure for the Queue 
struct Command { 
  char text[20]; 
  int blinkRate; 
}; 
// ========================================== 
// TASK 1: THE HEART (Blink LED) 
// ========================================== 
void HeartTask(void *pvParameters) { 
  pinMode(LED_PIN, OUTPUT); 

  for (;;) { //infinte loop
    digitalWrite(LED_PIN, HIGH); 
    vTaskDelay(100 / portTICK_PERIOD_MS);   // LED ON for 100 ms

    digitalWrite(LED_PIN, LOW); 
    vTaskDelay(currentDelay / portTICK_PERIOD_MS); // OFF delay
  } 
} 
// ========================================== 
// TASK 2: THE EAR (Serial Input) 
// ========================================== 
void InputTask(void *pvParameters) { 
  Serial.begin(115200); 

  for (;;) { 
    if (Serial.available() > 0) { 
      String input = Serial.readStringUntil('\n'); 
      input.trim(); 
      StaticJsonDocument<200> doc; 
      DeserializationError error = deserializeJson(doc, input); 
      if (!error) { 
        Command cmd; 

        strlcpy(cmd.text, doc["msg"] | "NoMsg", sizeof(cmd.text)); // Extract JSON data
        cmd.blinkRate = doc["delay"] | 1000; 

        // Send to Queue
        xQueueSend(commandQueue, &cmd, portMAX_DELAY); 
        Serial.println("Command sent to queue!"); 
      } 
      else { 
        Serial.println("JSON Error"); 
      } 
    }
    vTaskDelay(50 / portTICK_PERIOD_MS); 
  } 
} 
// ========================================== 
// TASK 3: THE FACE (OLED Display) 
// ========================================== 
void DisplayTask(void *pvParameters) { 
  Command receivedCmd; 
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {  
    for (;;); 
  } 
  // Initial Screen 
  display.clearDisplay(); 
  display.setTextSize(2); 
  display.setTextColor(SSD1306_WHITE); 
  display.setCursor(0, 20); 
  display.println("Waiting..."); 
  display.display(); 
  for (;;) { 
    // Wait indefinitely for Queue data
    if (xQueueReceive(commandQueue, &receivedCmd, portMAX_DELAY)) { 

      // Update global delay for Heart task
      currentDelay = receivedCmd.blinkRate; 

      // Update display
      display.clearDisplay(); 
      display.setCursor(0, 20); 
      display.println(receivedCmd.text); 
      display.display(); 

      Serial.println("Screen Updated."); 
    } 
  } 
} 
// ========================================== 
// SETUP 
// ========================================== 
void setup() { 
  // Create Queue (5 elements)
  commandQueue = xQueueCreate(5, sizeof(Command)); 

  // Create Tasks
  xTaskCreate(InputTask, "Ear", 4096, NULL, 1, &t_Input); 
  xTaskCreate(DisplayTask, "Face", 4096, NULL, 1, &t_Display); 
  xTaskCreate(HeartTask, "Heart", 2048, NULL, 2, &t_Blink); 
} 
void loop() {}
